def process_delete
  @id = params[:id]
  @user_toDelete = User.find(@id)
  if @user_toDelete.bookings.any?
    @user_toDelete.bookings.destroy_all
  end
  flash[:info] = "Account removed successfully"
  if @user_toDelete == @current_user
    @user_toDelete.destroy
    process_logout
  else
    @user_toDelete.destroy
    redirect_to :action => "view"
  end
end

def process_new
  @newUser = User.new(params[:user])
  unless params[:user][:admin]
    @newUser.admin = false
  end
  if @newUser.save
    MailUser.welcome(@newUser).deliver
    if @current_user
      flash[:message] = "New user created successfully"
      redirect_to root_url
    else
      flash[:message] = "Your account has been created, and you have been logged in"
      session[:id] = @newUser.id
      redirect_to root_url
    end
  else
    render :action => "new"
  end
end

def process_password
  if BCrypt::Engine.hash_secret(params[:oldPass], @current_user.password_salt) != @current_user.password_hash
    flash[:error] = "Your current password was incorrect"
    render :action => "password"
  elsif params[:newPass] != params[:newPassCheck]
    flash[:error] = "Your new passwords didn't match"
    render :action => "password"
  else
    flash[:message] = "Password changed successfully"
    @current_user.password = params[:newPass]
    @current_user.save
    redirect_to :action => "account"
  end
end

def process_reset
  if User.find_by_username(params[:user][:username])
    @user = User.find_by_username(params[:user][:username])
    @newPass = SecureRandom.hex(8)
    @user.password = @newPass
    @user.save
    MailUser.passreset(@user.id, @newPass).deliver
    flash[:message] = "Password reset successfully, please check your emails for your new password"
    redirect_to root_url
  else
    flash[:error] = "An account with that username was not found"
    render :action => "reset"
  end
end
