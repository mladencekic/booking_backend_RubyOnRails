def process_delete
  @room_toDelete = Room.find(params[:id])
  if @room_toDelete.bookings.any?
    @room_toDelete.bookings.destroy_all
  end
  @room_toDelete.destroy
  flash[:message] = "Room removed succesfully"
  redirect_to :action => "view"
end

def process_edit
  @room_toEdit = Room.find(params[:id])
  if @room_toEdit.update_attributes(params[:room])
    flash[:message] = "Room updated succesfully"
    redirect_to :action => "view"
  else
    render :action => "edit", :id => params[:id]
  end
end
