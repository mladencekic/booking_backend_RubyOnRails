def check_taken(currentBooking, existingBooking)

  @currentBooking = currentBooking
  @existingBooking = existingBooking

  if @currentBooking.start == @existingBooking.start
    @taken = true
  elsif @currentBooking.finish == @existingBooking.finish
    @taken = true
  elsif @currentBooking.finish > @existingBooking.start && @currentBooking.start < @existingBooking.start
    @taken = true
  elsif @currentBooking.start < @existingBooking.finish && @currentBooking.finish > @existingBooking.start
    @taken = true
  end
end

def process_edit
  @booking_toEdit = Booking.find(params[:id])
  if @booking_toEdit.update_attributes(params[:booking])
    MailUser.confirm_booking(@booking_toEdit.id, @booking_toEdit.user_id).deliver
    flash[:message] = "Edit was successful"
    redirect_to :action => "view", :view => "own"
  else
    render :action => "edit"
  end
end

def process_new
  @newBooking = @current_user.bookings.new(params[:booking])
  if @current_user.local
    @newBooking.total_price = ((Room.find(@newBooking.room_id).base_rate/100.0)*80).to_i
    if @newBooking.bar
      @newBooking.total_price = (@newBooking.total_price + 50).to_i
    end
  else
    @newBooking.total_price = Room.find(@newBooking.room_id).base_rate.to_i
    if @newBooking.bar
      @newBooking.total_price = (@newBooking.total_price + 50).to_i
    end
  end
  if Booking.find_by_room_id(@newBooking.room_id)
    @bookings = Booking.find_by_room_id(@newBooking.room_id)
    begin
      @bookings.each do |oldBooking|
        check_taken(@newBooking, oldBooking)
      end
    rescue NoMethodError
      check_taken(@newBooking, @bookings)
    end
  end
  if @taken
    flash[:error] = "The chosen time has already been taken"
    render :action => "new"
  elsif !@current_user.over_18 && @newBooking.bar
    flash[:error] = "Under 18s may not use the bar facilities"
    render :action => "new"

  elsif @newBooking.start.past? || @newBooking.finish.past?
    flash[:error] = "You can't create a booking for past dates"
    render :action => "new"

  elsif @newBooking.finish < @newBooking.start
    flash[:error] = "Your finish date can't be before the booking starts"
    render :action => "new"          

  elsif @newBooking.save
    MailUser.confirm_booking(@newBooking.id, @newBooking.user_id).deliver
    flash[:message] = "New booking created successfully"
    redirect_to :action => "view", :view => "own"

  else
    render :action => "new", :cost => @newBooking.total_price
  end
end

def view
  if params[:view] == "own"
    @bookings = @current_user.bookings
    @title = "Viewing Your Bookings"
    @own = true
  elsif params[:view] == "all"
    @bookings = Booking.find(:all)
    @title = "Viewing All Bookings"
    @own = false
  end
end
